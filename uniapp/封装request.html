export function apiRequest(options) {
    return new Promise((resolve, reject) => {
        const requestOptions = {
            url: options.url,
            method: options.method || "GET",
            data: options.data || {},
            header: options.header || {},
            success: (res) => {
                if (res.statusCode === 200) {
                    resolve(res.data);
                } else {
                    reject(res);
                }
            },
            fail: (err) => {
                reject(err);
            },
        };
        uni.request(requestOptions);
    });
}
